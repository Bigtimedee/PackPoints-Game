Create: /shared/cardhedge/types.ts

Define ONLY the fields PackPTS needs to display a card in the game:

export type Category =
  | "Baseball" | "Basketball" | "Football" | "Hockey" | "Soccer" | "Pokemon" | string;

export type CardSearchCard = {
  card_id: string;
  category: Category;
  category_group?: string;
  description: string;
  player: string;
  set: string;
  set_type?: string;
  number?: string;
  variant?: string;
  rookie?: boolean;
  image: string;            // raw from CardHedge
};

export type CardSearchRequest = {
  search?: string | null;
  set?: string | null;
  category?: Category | null;
  player?: string | null;
  rookie?: boolean | "Rookie" | null;
  raw_images_only?: boolean | null;
  page?: number;
  page_size?: number;
};

export type CardSearchResponse = {
  pages: number;
  count: number;
  cards: CardSearchCard[];
};

export type ImageSearchRequest = {
  image_url?: string | null;
  image_base64?: string | null;
  k?: number;               // 1..50 default 10
};

export type CardData = {
  card_id: string;
  description?: string;
  player?: string;
  set?: string;
  number?: string;
};

export type ImageSearchResultItem = {
  similarity: string;       // e.g. "95.23"
  distance: number;         // 0..1
  ximilar_id: string;
  product_id?: string | null;
  card_data?: CardData | null;
  // ai_fallback_info can exist but we will not depend on it for gameplay
};

export type ImageSearchResponse = {
  success: boolean;
  results: ImageSearchResultItem[];
  total_results: number;
  query_id: string;
  has_cardhedge_matches: boolean;
  message?: string | null;
};

export type CardDetailsRequest = {
  card_id: string;
  raw_images_only?: boolean | null;
};

export type CardDetails = {
  cards: CardSearchCard[];  // CardHedge returns cards[]; we reuse CardSearchCard fields
};

IMPORTANT:
- Do NOT include prices fields in any types.
- If CardHedge returns extra keys, ignore them in normalization.

D) CARDHEDGE CLIENT (SERVER ONLY)
Create: /server/integrations/cardhedge/client.ts

Implement:
- postCardHedge<T>(path: string, body: any): Promise<T>
  - uses fetch
  - headers: Content-Type + X-API-Key
  - timeout 8s via AbortController
  - retries on 502/503/429 up to 3 attempts with backoff (250ms, 750ms, 1500ms)

3) imageSearch(body: ImageSearchRequest): Promise<ImageSearchResponse>
  - POST /v1/cards/image-search
4) cardDetails(body: CardDetailsRequest): Promise<CardDetails>
  - POST /v1/cards/card-details

ALWAYS DEFAULT:
- raw_images_only = true unless caller explicitly passes false.

E) SERVER ROUTES (EXPRESS)
Create: /server/routes/cardhedge.routes.ts

Add these routes:

POST /api/cardhedge/card-search
Body: CardSearchRequest
Rules:
- Require at least one of: search, set, category, player
- page default 1
- page_size default 50, clamp 1..100
- raw_images_only default true
Response:
{ ok:true, data: CardSearchResponse } where cards are normalized and placeholder-filtered.

POST /api/cardhedge/search-cards-wsort
Body: { search?, set?, category?, player?, raw_images_only?, page?, page_size?, sort_by?, sort_order? }
Rules:
- Require at least one of: search, set, category, player (CardHedge requires this)
- clamp page/page_size
- raw_images_only default true
Response:
{ ok:true, data: CardSearchResponse } normalized + placeholder-filtered.

POST /api/cardhedge/image-search
Body: ImageSearchRequest
Rules:
- Require image_url OR image_base64
- k default 10, clamp 1..50
- if image_base64 present: reject if >10MB (413) BEFORE calling upstream
Response:
{ ok:true, data: ImageSearchResponse } (no pricing), and additionally return:
- best_match?: { card_id, similarity } if card_data exists and similarity >= 85

POST /api/cardhedge/card-details
Body: CardDetailsRequest
Rules:
- require card_id
- raw_images_only default true
Response:
{ ok:true, data: CardDetails } normalized; placeholder cards removed.

Error response format everywhere:
{ ok:false, error:{ code:string, message:string, status?:number } }

Wire router into Express app.
